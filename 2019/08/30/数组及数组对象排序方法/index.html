数组及数组对象排序方法

2019/08/30

//数组方法排序:

function sortNumber(a, b) {

return a - b;

}

//数组对象方法排序:

function sortByKey(array, key) {

return array.sort(function(a, b) {

var x = a[key];

var y = b[key];

return x < y ? -1 : x > y ? 1 : 0;

});

}
